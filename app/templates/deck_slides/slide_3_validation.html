Architectural Context: Before the unstructured JSON enters our data warehouse, it is strictly typed using a Pydantic schema model. We enforce domain rules (e.g., negative prices are impossible). Bad rows are quarantined. 
class DailyPriceRecord(BaseModel):
    price_date: date
    open_price: Decimal
    high_price: Decimal
    low_price: Decimal
    close_price: Decimal
    volume: int

    # Inline Pipeline Shielding
    @model_validator(mode="after")
    def validate_price_bounds(self):
        if any(p <= 0 for p in [self.open_price, self.high_price, self.close_price]):
            raise ValueError("QA Alert: Prices must be positive.")
        return self
