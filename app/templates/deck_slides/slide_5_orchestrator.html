Orchestration: Timeout Prevention

app/core/scheduler.py
Architectural Context: The Cron engine manages API state explicitly. It implements Rate Limiting (asyncio.sleep) between ticker pulls to prevent vendor IP blocking. Once finished, it pings an external Deadman's Switch for high-availability operational monitoring. 
async def sync_daily_prices():
    stocks = await StockRepository.fetch_stocks_needing_price_sync()
    
    if not stocks:
        # Monitor Observability: Alert if job dies quietly
        await ping_deadmans_switch() 
        return
        
    for stock in stocks:
        await StockIngestionService.process_daily_prices(stock["ticker"])
        
        # Operational Safety: Rate limit protection to prevent IP bans
        await asyncio.sleep(15)
